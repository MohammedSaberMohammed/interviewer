## What is the CDK?

The Angular Component Dev Kit (CDK) separates reusable, framework-level UI patterns from Angular Material's opinionated styling. You can use CDK to:
- Build your own component library with your design system
- Add complex behaviors (drag-drop, virtual scroll, overlays) without Material theming
- Implement accessibility patterns correctly without reinventing them

```bash
npm install @angular/cdk
```

## Drag and Drop

```typescript
import { DragDropModule, CdkDragDrop, moveItemInArray, transferArrayItem } from '@angular/cdk/drag-drop';

@Component({
  selector: 'app-kanban',
  standalone: true,
  imports: [DragDropModule],
  template: `
    <div class="kanban-board">
      @for (column of columns; track column.id) {
        <div class="column">
          <h3>{{ column.title }}</h3>

          <div cdkDropList
               [id]="column.id"
               [cdkDropListData]="column.items"
               [cdkDropListConnectedTo]="getConnectedLists(column.id)"
               (cdkDropListDropped)="drop($event)">

            @for (item of column.items; track item.id) {
              <div class="card" cdkDrag>
                <h4>{{ item.title }}</h4>
                <p>{{ item.description }}</p>
              </div>
            }

          </div>
        </div>
      }
    </div>
  `
})
class KanbanComponent {
  columns: KanbanColumn[] = [
    { id: 'todo', title: 'To Do', items: [...] },
    { id: 'in-progress', title: 'In Progress', items: [] },
    { id: 'done', title: 'Done', items: [] },
  ];

  getConnectedLists(currentId: string): string[] {
    return this.columns.map(c => c.id).filter(id => id !== currentId);
  }

  drop(event: CdkDragDrop<KanbanItem[]>) {
    if (event.previousContainer === event.container) {
      // Reorder within same list
      moveItemInArray(event.container.data, event.previousIndex, event.currentIndex);
    } else {
      // Move between lists
      transferArrayItem(
        event.previousContainer.data,
        event.container.data,
        event.previousIndex,
        event.currentIndex
      );
    }
  }
}
```

## Overlay: Custom Dropdown / Popover

The Overlay module lets you create positioned floating elements:

```typescript
import { Overlay, OverlayRef, OverlayConfig } from '@angular/cdk/overlay';
import { ComponentPortal } from '@angular/cdk/portal';

@Component({
  selector: 'app-dropdown-trigger',
  template: `<button (click)="openDropdown()" #triggerRef>Open Menu</button>`
})
class DropdownTriggerComponent implements OnDestroy {
  @ViewChild('triggerRef') triggerRef!: ElementRef;

  private overlay = inject(Overlay);
  private overlayRef: OverlayRef | null = null;

  openDropdown() {
    if (this.overlayRef?.hasAttached()) return;

    // Configure the overlay position
    const positionStrategy = this.overlay
      .position()
      .flexibleConnectedTo(this.triggerRef)
      .withPositions([
        { originX: 'start', originY: 'bottom', overlayX: 'start', overlayY: 'top' },
        { originX: 'start', originY: 'top', overlayX: 'start', overlayY: 'bottom' }, // Flip up if no space
      ]);

    this.overlayRef = this.overlay.create({
      positionStrategy,
      scrollStrategy: this.overlay.scrollStrategies.reposition(),
      hasBackdrop: true,
      backdropClass: 'cdk-overlay-transparent-backdrop',
    });

    // Render the dropdown component inside the overlay
    const portal = new ComponentPortal(DropdownMenuComponent);
    this.overlayRef.attach(portal);

    // Close on backdrop click
    this.overlayRef.backdropClick().subscribe(() => this.closeDropdown());
  }

  closeDropdown() {
    this.overlayRef?.dispose();
    this.overlayRef = null;
  }

  ngOnDestroy() {
    this.closeDropdown();
  }
}
```

## Portal: Render Content Elsewhere in the DOM

```typescript
import { PortalModule, TemplatePortal, CdkPortalOutlet } from '@angular/cdk/portal';

@Component({
  selector: 'app-layout',
  imports: [PortalModule],
  template: `
    <!-- Portal outlet in the header -->
    <header>
      <nav>...</nav>
      <div class="header-actions">
        <ng-template [cdkPortalOutlet]="headerActionsPortal"></ng-template>
      </div>
    </header>

    <main>
      <router-outlet></router-outlet>
    </main>
  `
})
class LayoutComponent {
  headerActionsPortal: TemplatePortal | null = null;

  setHeaderActions(portal: TemplatePortal) {
    this.headerActionsPortal = portal;
  }
}

// Child component projects content into the header via portal
@Component({
  imports: [PortalModule],
  template: `
    <ng-template #headerActions>
      <button (click)="save()">Save</button>
      <button (click)="cancel()">Cancel</button>
    </ng-template>

    <!-- Main content... -->
  `
})
class EditPageComponent implements OnInit {
  @ViewChild('headerActions', { read: TemplateRef })
  headerActionsTemplate!: TemplateRef<unknown>;

  private layout = inject(LayoutComponent);
  private viewContainerRef = inject(ViewContainerRef);

  ngOnInit() {
    const portal = new TemplatePortal(this.headerActionsTemplate, this.viewContainerRef);
    this.layout.setHeaderActions(portal);
  }
}
```

## Focus Trap: Accessible Modals

```typescript
import { A11yModule, FocusTrap, FocusTrapFactory } from '@angular/cdk/a11y';

@Component({
  selector: 'app-modal',
  imports: [A11yModule],
  template: `
    <div class="modal-overlay" role="dialog" aria-modal="true"
         [attr.aria-labelledby]="titleId">
      <div class="modal-content" cdkTrapFocus cdkTrapFocusAutoCapture>
        <h2 [id]="titleId">{{ title }}</h2>
        <ng-content></ng-content>
        <button (click)="close()">Close</button>
      </div>
    </div>
  `
})
class ModalComponent {
  @Input() title = '';
  titleId = `modal-title-${Math.random().toString(36).slice(2)}`;

  // cdkTrapFocus: Tab key won't leave the modal
  // cdkTrapFocusAutoCapture: Automatically focuses first focusable element
}
```

## Senior Insight: CDK vs Material vs Custom

| Use Case | Recommendation |
|----------|----------------|
| Rapid internal tooling | Angular Material (styled, complete) |
| Custom design system | CDK (behavior without style) |
| Performance-critical list | CDK Virtual Scroll |
| Accessible modal | CDK Focus Trap + ARIA attributes |
| Drag and drop board | CDK DragDrop |
| Tooltip on any element | CDK Overlay (or attribute directive) |

CDK's value: it provides correctly-implemented, accessible, tested implementations of complex interaction patterns. Building drag-and-drop from scratch with correct keyboard navigation, screen reader support, and cross-browser compatibility takes weeks. CDK provides it in minutes. Knowing when to leverage CDK vs build from scratch demonstrates architectural maturity.
